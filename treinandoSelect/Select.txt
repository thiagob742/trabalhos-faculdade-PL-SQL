
SELECT * 
FROM CLIENTE;

SELECT DESCRICAO NOME_PRODUTO
FROM PRODUTO;

SELECT P.*, C.NOME NOME_CLIENTE
FROM PEDIDO P INNER JOIN CLIENTE C
ON (P.CODCLIENTE = C.CODCLIENTE)
WHERE P.VALORTOTAL > 50;

SELECT * 
FROM PEDIDO
WHERE VALORTOTAL < 100 OR VALORTOTAL > 500;

SELECT *
FROM PEDIDO
WHERE DATAPEDIDO = '01/04/2012';

SELECT *
FROM PEDIDO
WHERE DATAPEDIDO = '01/04/2012'
AND VALORTOTAL BETWEEN 100 AND 200;

SELECT *
FROM PRODUTO
WHERE DESCRICAO LIKE '%M%'
OR DESCRICAO LIKE '%M%';

SELECT DISTINCT P.*, I.VALORUNITARIO
FROM PRODUTO P INNER JOIN ITEMPEDIDO I
ON (P.CODPRODUTO = I.CODPRODUTO)
WHERE I.VALORUNITARIO < 20;

SELECT P.CODPRODUTO CÓDIGO_PRODUTO, P.DESCRICAO NOME_PRODUTO, MIN(I.VALORUNITARIO) VALOR_MINIMO, MAX(I.VALORUNITARIO) VALOR_MÁXIMO
FROM PRODUTO P INNER JOIN ITEMPEDIDO I
ON(P.CODPRODUTO = I.CODPRODUTO)
GROUP BY P.CODPRODUTO, P.DESCRICAO, I.VALORUNITARIO;

SELECT C.NOME AS NOME_CLIENTE,
  C.CODCLIENTE AS CÓDIGO_CLIENTE,
  P.DATAPEDIDO AS DATA_PEDIDO
FROM CLIENTE C INNER JOIN PEDIDO P
ON (C.CODCLIENTE = P.CODCLIENTE)
ORDER BY 3 DESC;

SELECT DISTINCT C.NOME, COUNT(P.CODPEDIDO)
FROM CLIENTE C INNER JOIN PEDIDO P
ON (C.CODCLIENTE = P.CODCLIENTE)
GROUP BY C.NOME, 2;

SELECT MIN(CODCLIENTE) MENOR_CÓDIGO,
  MAX(CODCLIENTE) MAIOR_CÓDIGO
FROM CLIENTE;

SELECT MIN(VALORTOTAL) MENOR_VALOR_PEDIDO,
  MAX(VALORTOTAL) MAIOR_VALOR_PEDIDO
FROM PEDIDO;

SELECT SUM(VALORTOTAL) SOMA_VALORES_PEDIDOS
FROM PEDIDO;
SELECT SUM(VALORTOTAL) SOMA_VALORES_PEDIDOS
FROM PEDIDO
WHERE DATAPEDIDO BETWEEN '01/04/2012' 
  AND '30/04/2012';

SELECT ROUND(AVG(VALORTOTAL),2) SOMA_VALORES_PEDIDOS
FROM PEDIDO
WHERE EXTRACT(YEAR FROM DATAPEDIDO) = 2012; -- EXTRAI O ANO DA DATA ---

SELECT P.CODPRODUTO CÓDIGO_PRODUTO,
  P.DESCRICAO NOME_PRODUTO, 
  COUNT(I.QUANTIDADE)/COUNT(P.QUANTIDADE) MÉDIA_POR_QUANTIDADE
FROM PRODUTO P INNER JOIN ITEMPEDIDO I
ON(P.CODPRODUTO = I.CODPRODUTO)
GROUP BY P.CODPRODUTO, P.DESCRICAO, I.QUANTIDADE;

SELECT DISTINCT C.CODCLIENTE CÓDIGO_CLIENTE, 
  C.NOME NOME_CLIENTE
FROM PEDIDO P INNER JOIN CLIENTE C
ON (P.CODCLIENTE = C.CODCLIENTE)
ORDER BY 2 DESC;

SELECT P.CODPRODUTO CÓDIGO_PRODUTO, 
  P.DESCRICAO NOME_PRODUTO,
  I.VALORUNITARIO*I.QUANTIDADE VALOR_TOTAL_PRODUTO
FROM PRODUTO P INNER JOIN ITEMPEDIDO I
ON (P.CODPRODUTO = I.CODPRODUTO)
ORDER BY 2 DESC;
